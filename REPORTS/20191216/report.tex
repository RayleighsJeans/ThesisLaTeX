    \begin{frame}{STRAHL comparisons}
        \begin{block}{Basic procedure}
                \begin{itemize}
                    \only<1>{%
                        \item[+]{%
                            two sets of 1D simulations modelled after particular discharge XPID: 20181010.032, incorporating 4 scenarios where $f_{rad}=P_{rad}/P_{H}=\langle0.33, 0.66, 0.9, 1.0\rangle$
                        }%
                        \item[+]{%
                            each point in time trace with individual Thomson scattering profiles of $n_{e}$ \& $T_{e}$ as input for STRAHL
                        }%
                        \item[+]{%
                            both sets distinguished by different decay lenghts, transport profiles of $D/v$ and $D$ as well as profile shapes from TS (measured vs. Gaussian process fit)
                        }%
                    }%
                    \only<2>{%
                        \item[+]{%
                            set No.1 of simulation \# 2-5 $\lambda_{n, T, e}=$\SI{3}{\centi\meter} in alignment with profiles across plasma islands; set No.2 \# 6-9 $\lambda_{n, T, e}=$\SI{0.5}{\centi\meter} for 'free' SOL regions
                        }%
                        \item[+]{%
                            first set with measured and corrected data points from TS; second set based off of Gaussian process fits from measured data with most likely profile shape
                        }%
                        \item[+]{%
                            set No.1 constant $D=$\SI{0.5}{\square\meter\per\second} and null $D/v$ profile; set No.2 constant $D=$\SI{2.0}{\square\meter\per\second} and conservative electron root $D/v$ profile (see later)
                        }%
                    }%
                    \only<3>{%
                        \item[+]{%
                            otherwise equal $T_{i}$ profiles (none specified), neutral/pressure $n_{n}$, $p$ profiles (none given) and magnetic configurations (standard EJM)
                        }%
                        \item[+]{%
                            also both constant impurity sources located at $r_{LCFS}$ with intensity of \SI{3e17}{\per\second}
                        }%
                        \item[+]{%
                            same connections lengths to LCFS, divertor/limiter for both sets
                        }
                        \item[+]{%
                            relaxation time in simulation \SI{1}{\second}, convergence steps per time 1000
                        }%
                    }%
                \end{itemize}%
        \end{block}
    \end{frame}
