\usepackage{chemmacros}

%%% settings
\usechemmodule{all}


% Important note: chemmacros loads mathtools, which redefines over- and underbraces in a way it collides with unicode-math
% Therefore one has to use the standard LaTeX definition, which is accomplished by the following lines
\let\underbrace\LaTeXunderbrace
\let\overbrace\LaTeXoverbrace

% brackets around counter
\renewtagform{reaction}[]{(}{)}

\AtBeginEnvironment{reactions}{\renewcommand{\theequation}{\thechapter.\textls[-100]{\textsc{\roman{equation}}}}}
\AtEndEnvironment{reactions}{\renewcommand{\theequation}{\thechapter.\Roman{equation}}}
