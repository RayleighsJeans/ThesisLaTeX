% allows to show marginnotes
\usepackage{marginnote}

%%% Util definitions
\DeclareDocumentCommand{\myMarginnote} % needs xparse
					{
				 		O{0cm} O{c} m 
				 	}{
				 		% needs marginnote package
						\marginnote
								{
									\ifthispageodd
											{
												\RaggedRight % needs ragged2e package
											}{
												\RaggedLeft % needs ragged2e package
											}
									\raisebox{#1}[#1]
											{
												\begin{minipage}[#2]{\marginparwidth}
													\RaggedRight % needs ragged2e package
													\color{\getMainColor}
													\lineskiplimit=-\maxdimen
													\normalfont\sffamily%\scriptsize
													#3\end{minipage}}
											}
					}
