% easy referencing, noabbrev avoids abbreviations
\usepackage
		[
			noabbrev
		]
		{cleveref}

%%% Util definitions
\crefname{line}{line}{lines}
\creflabelformat{line}{#2#1#3}

\crefname{reaction}{reaction}{reactions}
\creflabelformat{reaction}{#2(#1)#3}

% make cleveref aware that we have code, not a listing
\crefname{listing}{code}{codes}
